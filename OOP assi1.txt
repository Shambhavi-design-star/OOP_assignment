#include<iostream>
using namespace std;
class bank_account
{
    string name,atype;
    int acc;
    float balance;
    public:
    bank_account(string n,string at,int an, float b)
    {
        name=n;
        atype=at;
        acc=an;
        balance=b;
    }
    void accept();
    void display();
    void withdraw();
    void deposit();
};
void bank_account::accept()
{
    cout<<"enter the name: \n";
    cin>>name;
    cout<<"enter account type: \n";
    cin>>atype;
    cout<<"enter account number:\n ";
    cin>>acc;
    cout<<"enter balance of the customer: \n";
    cin>>balance;
        }
void bank_account::display()
{
    cout<<"name:\n"<<name;
    cout<<"\n type of account:\n"<<atype;
    cout<<"\n account number:\n"<<acc;
    cout<<"\n balance:\n"<<balance;
}
void bank_account::withdraw()
{
   float amount_w;
   cout<<" enter the amount to be withdrawn:\n";
   cin>>amount_w;
   balance=balance-amount_w;
   cout<<"balance\n"<<balance;
}
void bank_account::deposit()
{
   float amount_d; 
   cout<<"enter the amount to be deposited:\n";
   cin>>amount_d;
   balance=balance+amount_d;
   cout<<" BALANCE\n"<<balance;
}
int main()
{
     string n, at;
        int anum;

            do {

                cout << "4. Exit\n";
                cout << "Enter your choice: ";
                cin >> choice;
                switch(choice) {
                    case 1:
                        obj.display();

                    case 2:
                        obj.deposit();

                    case 3:

                    case 4:
